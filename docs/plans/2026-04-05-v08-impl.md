# godot_agent/runtime/validation_checks.py
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

@dataclass
class CheckResult:
    name: str
    status: str          # pass | warning | error
    summary: str
    details: str
    file: str | None = None

class ValidationSuite:
    """Run all validation checks once, cache results for shared use."""

    def __init__(self, project_root: str | Path, changed_files: set[str]):
        self.project_root = Path(project_root)
        self.changed_files = changed_files
        self._results: dict[str, CheckResult] | None = None

    async def run_all(self) -> dict[str, CheckResult]:
        if self._results is not None:
            return self._results
        self._results = {}
        # Run each check category
        for check_name in self._check_names():
            result = await self._run_check(check_name)
            self._results[check_name] = result
        return self._results

    def get(self, check_name: str) -> CheckResult | None:
        if self._results is None:
            return None
        return self._results.get(check_name)

    def _check_names(self) -> list[str]:
        return [
            "file_exists", "gdscript_lint", "tscn_validate",
            "scene_resources", "ui_layout", "audio_nodes",
            "project_consistency", "dependency_graph",
            "pattern_advisor",
        ]

    async def _run_check(self, name: str) -> CheckResult:
        # Each check delegates to existing validation functions
        # Implementation per check type follows existing quality_gate.py logic
        return CheckResult(name=name, status="pass", summary="", details="")
```

# Count key-color pixels: create mask via point(), count non-zero in histogram
def _count_remaining_key_pixels(img, key_color, tolerance):
    key_img = Image.new("RGB", img.size, key_color)
    diff = ImageChops.difference(img.convert("RGB"), key_img)
    gray = diff.convert("L")
    mask = gray.point(lambda p: 1 if p <= tolerance else 0)
    return sum(mask.histogram()[:1])  # count of 0-value (matching) pixels... wait

    # Simpler: threshold then count
    threshold = gray.point(lambda p: 0 if p <= tolerance else 255)
    total = img.size[0] * img.size[1]
    non_key = sum(threshold.histogram()[1:])  # non-zero = not key color
    return total - non_key  # remaining key pixels
```
